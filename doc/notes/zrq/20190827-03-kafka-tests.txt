#[user@Grerat]

        for topic in $(tail -n 4 /tmp/topiclist)
        do
            echo "---- ----"
            echo "Topic [${topic}]"

            cat /tmp/${topic:?}-offsets \
            | sed '
                s/\([^:]*\):\([^:]*\):\([^:]*\)/\3/
                ' \
            | awk '
                {sum += $1} END {print sum}
                '
        done
