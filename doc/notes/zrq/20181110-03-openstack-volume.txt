# -----------------------------------------------------
# Setup our volume configuration.
#[root@openstacker]

    fsdev=/dev/vdd
    fspath=/var/local/outer/kafka/data-02

    volname=data-02
    volsize=1024

# -----------------------------------------------------
# Create our volume mount script.
#[root@openstacker]

cat > /tmp/volume-init.sh << 'EOSH'

#---------------------------------------------------------------------
# Create a filesystem.

    echo ""
    echo "Creating filesystem"
    mkfs.btrfs \
        --force \
        "${fsdev:?}"

    echo ""
    echo "Checking filesystem"
    btrfs \
        filesystem show \
        "${fsdev:?}"

#---------------------------------------------------------------------
# Create our mount point.

    echo "Creating mount point"

    mkdir -p "${fspath:?}"
    touch "${fspath:?}/mount-failed"

#---------------------------------------------------------------------
# Add the volume to our FileSystemTABle.
# https://www.howtoforge.com/reducing-disk-io-by-mounting-partitions-with-noatime

    echo "Updating fstab"
   
    fsuuid=$(
        lsblk --noheadings --output UUID "${fsdev:?}"
        )

cat >> /etc/fstab << EOTAB
UUID=${fsuuid:?} ${fspath:?}    btrfs    defaults,noatime    0  0
EOTAB

#---------------------------------------------------------------------
# Mount the new volume.

    echo "Mounting new filesystem"

    mount "${fspath:?}"

#---------------------------------------------------------------------
# Check the new volume.

    echo ""
    echo "free"

    df -h "${fspath:?}"

    echo ""
    echo "lsblk"

    lsblk

EOSH

        # -----------------------------------------------------
        # List our Kafka nodes.
        #[root@openstacker]

            for ident in ${kfidents[@]}
            do
                getvminfo "${ident:?}"
                name=$(getvmname)
                echo "Name  [$name]"
                echo "Ident [${ident}]"
            done

        # -----------------------------------------------------
        # Create and attach a new volume to each of our Kafka nodes.
        #[root@openstacker]

            for vmident in ${kfidents[@]}
            do

                getvminfo "${vmident:?}"
                echo "Name  [$(getvmname)]"
                echo "Ident [${vmident}]"

                volident=$(
                    makevolume \
                       "$(getvmname)-${volname:?}" \
                       "${volsize:?}" 
                    )

                openstack \
                    server add volume \
                    "${vmident:?}" \
                    "${volident:?}" \
                    --device "${voldev:?}"

            done

        # -----------------------------------------------------
        # Login and run our init script on each of our nodes.
        #[root@openstacker]

            for vmident in ${kfidents[@]}
            do

                internals=(
                    $(eleanorinternal)
                    )
                ipv4=${internals[1]}

                echo "---- ---- ---- ----"
                getvminfo "${vmident:?}"
                echo "Name    [$(getvmname)]"
                echo "Ident   [${vmident}]"
                echo "Address [${ipv4}]"

                echo "
                    sudo -s
                    fsdev=${fsdev:?}
                    fspath=${fspath:?}
                    " \
                    | cat - /tmp/volume-init.sh \
                    | ssh ${phym_sshopts[@]} \
                        "${sshuser:?}@${ipv4:?}" \

            done
