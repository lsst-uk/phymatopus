        # Try using one of our Kafka nodes as a proxy to the other nodes.
        #[root@openstacker]

            proxyip=172.16.49.217

            for vmident in $(
                openstack \
                    server list \
                    --format json \
                | jq -r '.[] | .ID'
                )
                do
                    getvminfo "${vmident:?}"
                    echo "---- ----"
                    echo "Ident [$vmident]"
                    echo "Name  [$(getvmname)]"
                    echo "ip192 [$(geteleanor192)]"
                    echo ""
                    ssh ${sshopts[@]} \
                        -o ProxyCommand="ssh ${sshopts[*]} -W %h:%p ${sshuser:?}@${proxyip:?}" \
                        ${sshuser:?}@$(geteleanor192) \
                            "
                            hostname
                            "
                    echo ""
                done
