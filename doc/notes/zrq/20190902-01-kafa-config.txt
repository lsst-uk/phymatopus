# Login to each of the Kafka nodes and update the config.
#[user@trop03]

    #   1000    1ms  => 1sec
    #   *60     1sec => 1min
    #   *60     1min => 1hr
    #   *24     1hr  => 1day
    #   *7      1day => 7days

    retention=$((1000*60*60*24*7))

    #  4 physical
    # 16 virtual
    #  4 threads

    partitions=$((4*16*4))

    source "${HOME}/nodenames"
    source "${HOME}/ssh-options"

    for vmname in "${kfnames[@]}"
        do
            echo "---- ----"
            echo "Node [${vmname:?}]"
            ssh ${sshopts[@]} \
                "${sshuser:?}@${vmname:?}" \
                    "
                    date
                    hostname
                    sed -i '
                        /KAFKA_LOG_RETENTION_MS/ {
                            s/=.*\$/=${retention}\"/
                            }
                        /KAFKA_NUM_PARTITIONS/ {
                            s/=.*\$/=${partitions}\"/
                            }
                        ' \${HOME}/kafka.yml
                    "
        done
